## 📝 ملاحظات للمطورين

### حقل تاريخ الميلاد:
```tsx
// المكونات المستخدمة
import { DayPicker } from 'react-day-picker';
import { format } from 'date-fns';
import { ar } from 'date-fns/locale';

// الإعدادات الرئيسية
captionLayout="dropdown-buttons"  // قوائم منسدلة للشهر والسنة
fromYear={1950}                    // من سنة 1950
toYear={new Date().getFullYear() - 3}  // حتى قبل 3 سنوات
locale={ar}                        // اللغة العربية
dir="rtl"                          // اتجاه من اليمين لليسار
```
